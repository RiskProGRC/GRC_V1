Phase 2 — usersClass.php — New Methods

5 new methods to support the module features. Add below the existing methods in Project/users/usersClass.php.

M1

toggleAccess() — suspend / activate a user

New Method
// Flip access: 0→1 (activate) or 1→0 (suspend). Returns new state label.
public function toggleAccess(string $uid, string $adminUid, string $ip): string {
    $stmt = $this->prepare("UPDATE users SET access = 1 - access WHERE id = ?");
    $stmt->bind_param('s', $uid);
    $stmt->execute();
    if ($stmt->affected_rows < 1) return 'No change';
    $row = $this->fetchOne('users', 'id', $uid);
    $state = ($row['access'] == 1) ? 'activated' : 'suspended';
    ActivityLogger::log($this, $adminUid, 'Users', "$state user id=$uid", $ip);
    return $state;
}
M2

deleteUser() — soft delete

New Method
// Soft-delete: sets deleted_at. Requires Phase 7 DB migration first.
public function deleteUser(string $uid, string $adminUid, string $ip): bool {
    $stmt = $this->prepare("UPDATE users SET deleted_at = NOW() WHERE id = ? AND deleted_at IS NULL");
    $stmt->bind_param('s', $uid);
    $stmt->execute();
    if ($stmt->affected_rows < 1) return false;
    ActivityLogger::log($this, $adminUid, 'Users', "deleted user id=$uid", $ip);
    return true;
}
M3

adminResetPassword() — force password reset

New Method
// Sets a temp password and flags first_login=1 so the user must change it.
public function adminResetPassword(string $uid, string $tempPass, string $adminUid, string $ip): string {
    $hash = password_hash($tempPass, PASSWORD_DEFAULT);
    $stmt = $this->prepare("UPDATE users SET upassword = ?, first_login = 1 WHERE id = ?");
    $stmt->bind_param('ss', $hash, $uid);
    $stmt->execute();
    if ($stmt->affected_rows < 1) return 'Reset failed';
    ActivityLogger::log($this, $adminUid, 'Users', "force-reset password for user id=$uid", $ip);
    return 'Password reset. User must change on next login.';
}
M4

usernameExists() — duplicate check

New Method
// Returns true if username already taken (used by the live-check endpoint).
public function usernameExists(string $uname): bool {
    return $this->fetchOne('users', 'username', $uname) !== null;
}
M5

getStats() — dashboard aggregate

New Method
// Returns totals for the stats cards on userslist.php.
public function getStats(): array {
    $stmt = $this->prepare(
        "SELECT COUNT(*) AS total,
                SUM(access = 1) AS active,
                SUM(access = 0) AS suspended,
                SUM(roles  = 1) AS admins,
                SUM(roles  = 2) AS users
         FROM users
         WHERE deleted_at IS NULL"
    );
    $stmt->execute();
    return $stmt->get_result()->fetch_assoc() ?? [];
}
Requires deleted_at column from Phase 7. If you run this before migration, remove the WHERE clause temporarily.

Phase 6 — assets/js/usermanagement.js

New JS file. Include it only on user management pages. Covers all AJAX handlers for the module.

/* assets/js/usermanagement.js */
$(function () {

    /* ── Add User form submit ── */
    function validateForm() {
        var valid = true;
        ['fname','sname','uname','email','phone','password'].forEach(function(id) {
            var $el = $('#' + id);
            if (!$el.val().trim()) { $el.addClass('is-invalid'); valid = false; }
            else                   { $el.removeClass('is-invalid'); }
        });
        return valid;
    }

    $('.adduser-button').on('click', function() {
        if (!validateForm()) { $('#form-feedback').html('<div class="alert alert-warning">Please fill all required fields.</div>'); return; }
        var $btn = $(this).prop('disabled', true).text('Adding…');
        $.ajax({
            url: window.GRC.base + 'Project/usersaction.php',
            method: 'POST', dataType: 'json',
            data: $('#profileform').serialize(),
            success: function(res) {
                var cls = res.status === 'ok' ? 'success' : 'danger';
                $('#form-feedback').html('<div class="alert alert-'+cls+'">'+res.message+'</div>');
                if (res.status === 'ok') $('#profileform')[0].reset();
            },
            error: function() { $('#form-feedback').html('<div class="alert alert-danger">Server error.</div>'); },
            complete: function() { $btn.prop('disabled', false).text('ADD USER'); }
        });
    });

    /* ── Username live check ── */
    $('#uname').on('blur', function() {
        var val = $(this).val().trim(); if (!val) return;
        $.getJSON(window.GRC.base + 'Project/usernamecheck.php', { uname: val }, function(res) {
            $('#uname').toggleClass('is-invalid', res.taken).toggleClass('is-valid', !res.taken);
            $('#uname-feedback').text(res.taken ? 'Username already taken.' : '');
        });
    });

    /* ── Show / hide password ── */
    $('.toggle-pw').on('click', function() {
        var $inp = $('#password');
        var shown = $inp.attr('type') === 'text';
        $inp.attr('type', shown ? 'password' : 'text');
        $(this).attr('aria-label', shown ? 'Show password' : 'Hide password');
    });

    /* ── Password strength ── */
    $('#password').on('keyup', function() {
        var v = $(this).val(), s = 0;
        if (v.length >= 8) s++; if (/[A-Z]/.test(v)) s++; if (/\d/.test(v)) s++; if (/\W/.test(v)) s++;
        var c = ['','bg-danger','bg-warning','bg-info','bg-success'][s];
        var l = ['','Weak','Fair','Good','Strong'][s];
        $('#pw-bar').css('width', (s*25)+'%').removeClass().addClass('progress-bar '+c);
        $('#pw-label').text(l);
    });

    /* ── Access toggle (userslist.php) ── */
    $(document).on('click', '.access-toggle', function() {
        var $btn = $(this), tUid = $btn.data('uid'), csrf = $btn.data('csrf');
        $.ajax({
            url: window.GRC.base + 'Project/useraccessaction.php',
            method: 'POST', dataType: 'json',
            data: { uid: tUid, csrf_token: csrf },
            success: function(res) {
                if (res.status === 'ok') {
                    var active = res.message === 'activated';
                    $btn.text(active ? 'Active' : 'Suspended')
                        .removeClass('btn-success btn-danger')
                        .addClass(active ? 'btn-success' : 'btn-danger');
                } else { Swal.fire('Error', res.message, 'error'); }
            }
        });
    });

    /* ── Delete user (userslist.php) ── */
    $(document).on('click', '.user-delete', function() {
        var $btn = $(this), tUid = $btn.data('uid'), name = $btn.data('name'), csrf = $btn.data('csrf');
        Swal.fire({ title: 'Delete '+name+'?', text: 'This cannot be undone.', icon: 'warning',
            showCancelButton: true, confirmButtonColor: '#d33', confirmButtonText: 'Yes, delete'
        }).then(function(r) {
            if (!r.isConfirmed) return;
            $.ajax({
                url: window.GRC.base + 'Project/userdeleteaction.php',
                method: 'POST', dataType: 'json',
                data: { uid: tUid, csrf_token: csrf },
                success: function(res) {
                    if (res.status === 'ok') { $btn.closest('tr').fadeOut(function(){ $(this).remove(); }); }
                    else { Swal.fire('Error', res.message, 'error'); }
                }
            });
        });
    });

    /* ── Reset password (profile.php) ── */
    $(document).on('click', '.reset-pw-btn', function() {
        var tUid = $(this).data('uid'), csrf = $(this).data('csrf');
        Swal.fire({ title: 'Set temporary password', input: 'password',
            inputAttributes: { autocomplete: 'new-password', minlength: 8 },
            showCancelButton: true, confirmButtonText: 'Reset'
        }).then(function(r) {
            if (!r.isConfirmed || !r.value) return;
            $.ajax({
                url: window.GRC.base + 'Project/userresetpwaction.php',
                method: 'POST', dataType: 'json',
                data: { uid: tUid, new_pass: r.value, csrf_token: csrf },
                success: function(res) { Swal.fire(res.status === 'ok' ? 'Done' : 'Error', res.message, res.status === 'ok' ? 'success' : 'error'); }
            });
        });
    });

    /* ── Dept filter (userslist.php) ── */
    $('#dept-filter').on('change', function() {
        var val = $(this).val();
        $('#table1 tbody tr').show();
        if (val) { $('#table1 tbody tr').filter(function() { return $(this).data('dept') !== val; }).hide(); }
    });

});
